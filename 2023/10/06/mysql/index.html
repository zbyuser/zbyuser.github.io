基本sql操作

SQL语句：

操作文件夹（库）
增
create database db1 charset utf8;
查
show create database db1;
show databases;
改
alter database db1 charset gbk;
删
drop database db1;

操作文件（表）
切换文件夹：use db1;
查看当前所在文件夹：select database();

查
    show create table t1;
    show tables;
    desc t1;
改
    alter table t1 modify name char(6);
    alter table t1 change name NAME char(7);
删

增
insert t1(id,name) values(1,’egon1’),(2,’egon2’),(3,’egon3’);

select id,name from db1.t1;
select * from db1.t1;
改
update db1.t1 set name=’SB’;
update db1.t1 set name=’ALEX’ where id=2;
删
delete from t1;
delete from t1 where id=2;

存储引擎

1、什么是存储引擎？
存储引擎就是表的类型

2、查看MySQL支持的存储引擎
show engines;

3、指定表类型/存储引擎
create table t1(id int)engine=innodb; (默认) innodb在硬盘中对应两种文件(.frm存储表对的结构 .ibd存储记录)
create table t2(id int)engine=memory; （数据只存储在内存中，不存储到磁盘中）
create table t3(id int)engine=blackhole; （数据放进去就没了，相当于是一个黑洞）
create table t4(id int)engine=myisam;

insert into t1 values(1);
insert into t2 values(1);
insert into t3 values(1);
insert into t4 values(1);

字符类型

默认情况下char在存储的时候按照指定的字符的数量进行存储，如果不够字符的话，补充空格进行存储，但是在取出数据的时候又会去掉空格

char:定长
varchar：变长

#宽度指的是字符的个数
create table t13(name char(5));
create table t14(name varchar(5));

insert into t13 values(‘李杰 ‘); #’李杰 ‘ 5个字符（补充空格，存入数据为5个字符）
insert into t14 values(‘李杰 ‘); #’李杰 ‘ 3个字符

select char_length(name) from t13; #5 char_length函数获取字段的长度(注意提前设置sql_mode)
select char_length(name) from t14; #3

select name from t13 where name=’李杰’; # 在查询的时候会忽略末尾的空格进行查询
select name from t13 where name like ‘李杰’; # 查询不到结果。like ‘李杰 ‘ 就查询得到结果因为实际上存储到表中的数据为“李杰 ”

name char(5)
egon |alex |wxx |
取的时候5个5个字节取就行了（存储数据的速度快）

name varchar(5)
1bytes+egon|1bytes+alex|1bytes+wxx| 用1个bytes来表示字符的长度
4+egon|4+alex|3+wxx|
